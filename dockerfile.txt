# --- STAGE 1: The Collector (Ubuntu 20.04) ---
FROM ubuntu:20.04 AS collector
ENV DEBIAN_FRONTEND=noninteractive

RUN apt-get update && apt-get install -y \
    clang-format-11 \
    clang-tidy-11 \
    clang-tidy-9 \
    && rm -rf /var/lib/apt/lists/*

RUN mkdir -p /opt/legacy-clang/bin /opt/legacy-clang/lib

# Smart copy: Grab binaries and their specific dependencies
RUN for tool in clang-format-11 clang-tidy-11 clang-tidy-9; do \
    tool_path=$(which $tool); \
    cp $tool_path /opt/legacy-clang/bin/; \
    # We grab dependencies BUT exclude the core system Glibc/Loader 
    # to prevent version mismatch crashes on the host 22.04
    ldd $tool_path | grep "=> /" | grep -vE "libc.so|ld-linux|libpthread.so" | awk '{print $3}' | xargs -I '{}' cp -n '{}' /opt/legacy-clang/lib/; \
    done

# --- STAGE 2: Final Build Image (Ubuntu 22.04) ---
FROM my-base-build:latest

USER root

# 1. Install modern Clang-18 natively
RUN apt-get update && apt-get install -y clang-format-18 && rm -rf /var/lib/apt/lists/*

# 2. Copy the isolated legacy bundle
COPY --from=collector /opt/legacy-clang /opt/legacy-clang

# 3. Create "Safe Wrappers" in /usr/local/bin
# This allows you to call 'clang-tidy-9' normally, but it uses its own libs.
RUN for tool in clang-format-11 clang-tidy-11 clang-tidy-9; do \
    printf '#!/bin/sh\nLD_LIBRARY_PATH=/opt/legacy-clang/lib /opt/legacy-clang/bin/%s "$@"' "$tool" > /usr/local/bin/$tool; \
    chmod +x /usr/local/bin/$tool; \
    done

# Verify
RUN clang-format-18 --version && \
    clang-format-11 --version && \
    clang-tidy-9 --version
