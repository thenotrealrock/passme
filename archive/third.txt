# --- STAGE 1: The Collector (Ubuntu 20.04) ---
FROM ubuntu:20.04 AS collector
ENV DEBIAN_FRONTEND=noninteractive

# Define the tools once to prevent duplication
ENV TOOLS="clang-format-11 clang-tidy-11 clang-tidy-9"

RUN apt-get update && apt-get install -y $TOOLS patchelf && rm -rf /var/lib/apt/lists/*
RUN mkdir -p /opt/legacy-clang/bin /opt/legacy-clang/lib

# Single loop handles all tools identically
RUN for tool in $TOOLS; do \
    tool_path=$(which $tool); \
    base_name=$(basename $tool_path); \
    # Copy binary
    cp $tool_path /opt/legacy-clang/bin/; \
    # Copy all library dependencies
    ldd $tool_path | grep "=> /" | awk '{print $3}' | xargs -I '{}' cp -n '{}' /opt/legacy-clang/lib/; \
    # Copy the 20.04 loader (only needs to happen once, but loop-safe)
    cp /lib64/ld-linux-x86-64.so.2 /opt/legacy-clang/lib/; \
    # Patch binary to be self-contained
    patchelf --set-interpreter /opt/legacy-clang/lib/ld-linux-x86-64.so.2 /opt/legacy-clang/bin/$base_name; \
    patchelf --set-rpath /opt/legacy-clang/lib /opt/legacy-clang/bin/$base_name; \
    done

# --- STAGE 2: Final Build Image (Ubuntu 22.04) ---
FROM ubuntu:22.04
USER root
ENV DEBIAN_FRONTEND=noninteractive
ENV TOOLS="clang-format-11 clang-tidy-11 clang-tidy-9"

# 1. Install modern Clang-18
RUN apt-get update && apt-get install -y clang-format-18 && rm -rf /var/lib/apt/lists/*

# 2. Copy the isolated bundle
COPY --from=collector /opt/legacy-clang /opt/legacy-clang

# 3. Use a loop to create symlinks in the system PATH
RUN for tool in $TOOLS; do \
    ln -s /opt/legacy-clang/bin/$tool /usr/local/bin/$tool; \
    done

# Verify all tools in one block
RUN clang-format-18 --version && \
    for tool in $TOOLS; do $tool --version; done

# --- STAGE 1: The Collector (Ubuntu 20.04) ---
FROM ubuntu:20.04 AS collector
ENV DEBIAN_FRONTEND=noninteractive
ENV TOOLS="clang-format-11 clang-tidy-11 clang-tidy-9"

RUN apt-get update && apt-get install -y $TOOLS patchelf binutils && rm -rf /var/lib/apt/lists/*
RUN mkdir -p /opt/legacy-clang/bin /opt/legacy-clang/lib

RUN for tool in $TOOLS; do \
    tool_path=$(which $tool); \
    base_name=$(basename $tool_path); \
    cp $tool_path /opt/legacy-clang/bin/; \
    # 1. Automated Loader Detection: Find the system path of the required loader
    loader_path=$(readelf -l $tool_path | grep 'interpreter' | sed -e 's/.*: \(.*\)]/\1/'); \
    # 2. Copy the loader to our local lib
    cp "$loader_path" /opt/legacy-clang/lib/; \
    # 3. Copy all library dependencies
    ldd $tool_path | grep "=> /" | awk '{print $3}' | xargs -I '{}' cp -n '{}' /opt/legacy-clang/lib/; \
    # 4. Patch: Point to the local loader (relative to the final /opt path) and local libs
    patchelf --set-interpreter "/opt/legacy-clang/lib/$(basename $loader_path)" "/opt/legacy-clang/bin/$base_name"; \
    patchelf --set-rpath /opt/legacy-clang/lib "/opt/legacy-clang/bin/$base_name"; \
    done

# --- STAGE 2: Final Build Image (Ubuntu 22.04) ---
FROM ubuntu:22.04
USER root
ENV TOOLS="clang-format-11 clang-tidy-11 clang-tidy-9"

RUN apt-get update && apt-get install -y clang-format-18 && rm -rf /var/lib/apt/lists/*
COPY --from=collector /opt/legacy-clang /opt/legacy-clang

# Automatically symlink everything in the legacy bin folder
RUN for tool in $TOOLS; do ln -s /opt/legacy-clang/bin/$tool /usr/local/bin/$tool; done

# Verify
RUN clang-format-18 --version && \
    for tool in $TOOLS; do $tool --version; done
